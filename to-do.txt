2- porta
install 
    npm i homebridge-doorsensor
config.json
    {
        "accessory": "DoorSensor",
        "name": "Garage Door",
        "sensorPollInMs": 5000,
        "flipBehavior": false,
        "sensorPin": 18,         /* sensorPin or triggerFile required */
        "triggerFile": "file"    /* but not both */
    }



"accessories": [
        {
            "accessory": "Humidity Sensor",
            "name": "Pi Humidity Sensor",
            "pin": 4
        },
        {
            "accessory": "DoorSensor",
